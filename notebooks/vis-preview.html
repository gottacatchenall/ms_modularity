function assortative(;
    num_nodes = 50,
    num_blocks = 5,
    within_dist = Uniform(0.5,0.8),
    between_dist = Uniform(0,0.1),
    disassortative = false
)
    nodes_per_block = num_nodes ÷ num_blocks
    prob_mat = zeros(num_nodes, num_nodes)

    block_labels = vcat([[i for _ in 1:nodes_per_block] for i in 1:num_blocks]...)

    for i in 1:num_nodes, j in 1:num_nodes
        op = !disassortative ? isequal : !isequal
        dist = op(block_labels[i], block_labels[j]) ? within_dist : between_dist
        prob_mat[i,j] = rand(dist)
        prob_mat[j,i] = rand(dist)
    end
    prob_mat
end
